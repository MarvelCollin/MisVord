server {
    # SSL Configuration managed by Certbot
    listen 443 ssl http2;
    server_name marvelcollin.my.id www.marvelcollin.my.id;
    ssl_certificate /etc/letsencrypt/live/marvelcollin.my.id/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/marvelcollin.my.id/privkey.pem;
    
    # Security headers
    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header X-XSS-Protection "1; mode=block" always;
    add_header Referrer-Policy "no-referrer-when-downgrade" always;
    add_header Content-Security-Policy "default-src 'self' https: data: 'unsafe-inline' 'unsafe-eval'" always;

    # Static files for CSS, JS, Images
    location /public/ {
        alias /home/kolin/MisVord/App/public/;
        expires 1y;
        add_header Cache-Control "public, immutable";
        try_files $uri =404;
    }

    location /socket-health {
        proxy_pass http://127.0.0.1:1002/socket-health;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }


    # Aturan untuk Socket.IO
    location /socket.io/ {
        proxy_pass http://127.0.0.1:1002;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
        proxy_read_timeout 86400;
    }

    # Aturan utama untuk aplikasi PHP (sebagai fallback)
    location / {
        proxy_pass http://localhost:1001;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
}
server {
    if ($host = www.marvelcollin.my.id) {
        return 301 https://$host$request_uri;
    } # managed by Certbot

    if ($host = marvelcollin.my.id) {
        return 301 https://$host$request_uri;
    } # managed by Certbot

    listen 80;
    server_name marvelcollin.my.id www.marvelcollin.my.id;
    return 301 https://$server_name$request_uri;
}
