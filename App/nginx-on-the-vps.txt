server {
    # Bagian ini akan diisi otomatis oleh Certbot dengan pengaturan SSL
    listen 443 ssl;
    server_name marvelcollin.my.id www.marvelcollin.my.id;
    ssl_certificate /etc/letsencrypt/live/marvelcollin.my.id/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/marvelcollin.my.id/privkey.pem;
    # ... pengaturan SSL lainnya dari Certbot ...

    # Aturan BARU untuk file statis (CSS, JS, Gambar)
    location /misvord/public/ {
        # PENTING: Pastikan path ini benar-benar menuju folder proyek Anda
        root /home/kolin/MisVord/App;
        try_files $uri =404;
    }

    location /socket-health {
        proxy_pass http://127.0.0.1:1002/socket-health;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }


    # Aturan untuk Socket.IO
    location /socket.io/ {
        proxy_pass http://127.0.0.1:1002;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header "Connection";
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
        proxy_read_timeout 86400;
    }

    # Aturan utama untuk aplikasi PHP (sebagai fallback)
    location / {
        proxy_pass http://localhost:1001;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
}
server {
    if ($host = www.marvelcollin.my.id) {
        return 301 https://$host$request_uri;
    } # managed by Certbot

    if ($host = marvelcollin.my.id) {
        return 301 https://$host$request_uri;
    } # managed by Certbot

    listen 80;
    server_name marvelcollin.my.id www.marvelcollin.my.id;
    return 404; # managed by Certbot
}
